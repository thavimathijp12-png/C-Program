#include <stdio.h>

int main() {
    int classType, age;
    int fare;

    scanf("%d", &classType);
    scanf("%d", &age);

    switch (classType) {
        case 1:   
            fare = 300;

            if (age < 12) {
                fare = fare / 2;
            }
            else if (age >= 60) {
                fare = (fare * 2) / 3;
            }
            break;

        case 2:   // AC
            fare = 1000;

            if (age < 12) {
                fare = fare / 2;
            }
            break;

        default:
            printf("Invalid Class");
            return 0;
    }

    printf("Fare â‚¹%d", fare);
    return 0;
}
